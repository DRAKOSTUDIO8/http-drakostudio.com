    <script>
        function sendWhatsApp() {
            let name = document.getElementById("name").value;
            let address = document.getElementById("address").value;
            let total = document.getElementById("total").value;
            let orderNumber = document.getElementById("orderNumber").value;
            
            let phone = "+529211354623"; // Reemplaza con tu número de WhatsApp
            let message = `Hola, quiero confirmar mi pedido:\n\nNombre: ${name}\nDirección: ${address}\nTotal a pagar: $${total}\nNúmero de orden: ${orderNumber}`;
            
            let url = `https://wa.me/${phone}?text=${encodeURIComponent(message)}`;
            window.open(url, "_blank");
        }
